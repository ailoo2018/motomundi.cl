/**
** DEPLOY TO AZURE
**/

// login
az login

// create resource group in eastus
 az group create --name motomundi-rg --location eastus

// create container registry (like docker registry)
az acr create --resource-group motomundi-rg --name motomundiregistry --sku Basic

// loginin to container registry
az acr login --name motomundiregistry

// tag docker image
docker tag motomundi-app motomundiregistry.azurecr.io/motomundi-nuxt:v1

// upload tagged image to azure registry
docker push motomundiregistry.azurecr.io/motomundi-web:v1

// add web to  containerapp specifiying eastus
az containerapp up `
  --name motomundi-www `
  --resource-group motomundi-rg `
  --location eastus `
  --image motomundiregistry.azurecr.io/motomundi-nuxt:v1 `
  --ingress external `
  --target-port 3000
